import re

def split_words_and_separators(text):
    # Regex to split and capture the separators (space, comma + space)
    parts = re.split(r'(\s|,\s)', text)
    
    words = []
    separators = []

    i = 0
    while i < len(parts):
        if parts[i] not in [' ', ', ']:  # It's a word
            words.append(parts[i])
            # Check if next part is a separator
            if i + 1 < len(parts) and parts[i + 1] in [' ', ', ']:
                separators.append(parts[i + 1])
            i += 2
        else:
            i += 1  # Skip unexpected separator at start

    return [words, separators]


# Test examples
inputs = [
    "W3resource Python, Exercises.",
    "The dance, held in the school gym, ended at midnight.",
    "The colors in my studyroom are blue, green, and yellow."
]

for text in inputs:
    print(f"Input: {text}")
    print("Output:", split_words_and_separators(text))
    print()
